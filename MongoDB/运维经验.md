迁移脚本

```js
use udesk_applog;

var query = { _id: { $lt: ObjectId("6609f9000000000000000000") } };
var collection = db.getCollection('call_channel');

// 删除速率控制，每次删除一条文档并等待 1 秒
function deleteWithRateLimit() {
    var doc = collection.find(query).sort({ _id: 1 }).limit(1000);
    if (doc.hasNext()) {
        var docToDelete = doc.next();
        collection.deleteOne({ _id: docToDelete._id });
        sleep(1000); // 控制删除速率：等待 1 秒
        deleteWithRateLimit(); // 递归删除下一条文档
    } else {
        print('删除完成');
    }
}

// 开始删除操作
deleteWithRateLimit();


use udesk_applog;
function deleteByLimit() {
    var cursor = db.call_channel.find({ _id: { $lt: ObjectId("66aad0000000000000000000") } }, {_id: 1}).sort({_id: 1}).skip(1000).limit(1)
    if (cursor.hasNext()) {
        var obj = cursor.next()
        var curID = obj._id

        print(curID, "\n", curID.getTimestamp())
        var res = db.call_channel.deleteMany({"_id" : { $lt: curID }})
        printjson(res)
        sleep(500); // 控制删除速率：等待 1 秒

        deleteByLimit()
    }
}

var stat = db.call_channel.stats(); printjson({ "size" : stat.size, "count" : stat.count, "avgObjSize" : stat.avgObjSize, "storageSize" : stat.storageSize,})
```
