# conda

```sh
conda init # 新 shell 运行 conda 的时候，conda 初始化 shell 环境
conda create -n myenv python=3.10 # 创建环境，指定版本
conda env list
conda activate myenv              # 激活环境

# 安装依赖的三种方式
pip install -r requirements.txt   # 按 req.txt 安装依赖包
conda install django flask        # 手动安装依赖包
pip install django flask          # 使用 pip 安装依赖包

pip install -i https://mirrors.tuna.tsinghua.edu.cn/pypi/web/simple PACKAGENAME

# 启动项目的方式
python app.py                       # 脚本项目
python manage.py runserver          # django 项目
export FLASK_APP=app.py; flask run  # flask 项目
```
