# lk-cli

```sh
# 命令
   app       Initialize and manage applications
   agent, a  Manage LiveKit Cloud Agents
   cloud     Interact with LiveKit Cloud services
   project   Add or remove projects and view existing project properties
   room      Create or delete rooms and manage existing room properties
   token     Create access tokens with granular capabilities
   dispatch  Create, list, and delete agent dispatches
   egress    Record or stream media from LiveKit to elsewhere
   ingress   Import outside media sources into a LiveKit room
   sip       Manage SIP Trunks, Dispatch Rules, and Participants
   perf      Performance testing commands

# 全局 options
   --url URL              URL to LiveKit instance (default: "http://localhost:7880") [$LIVEKIT_URL]
   --api-key KEY          Your KEY [$LIVEKIT_API_KEY]
   --api-secret SECRET    Your SECRET [$LIVEKIT_API_SECRET]
   --dev                  Use developer credentials for local LiveKit server (default: false)
   --project NAME         NAME of a configured project
   --subdomain SUBDOMAIN  SUBDOMAIN of a configured project
   --config TOML          Config TOML to use in the working directory (default: "livekit.toml")
   --curl                 Print curl commands for API actions (default: false)
   --verbose              (default: false)
   --help, -h             show help
   --version, -v          print the version

# 使用
lk room list [RoomName]
        create
        update
        delete RoomName
        join
             --identity ID --attribute k1=v1 --attribute k2=v2  RoomName
             --identity ID --attribute-file FILE.json           RoomName
             --identity ID --publish-demo                       RoomName
             --identity ID --publish <a.ivf> --publish <b.ogg --fps 32.98   RoomName
             --identity ID --publish h264:///tmp/myvideo.sock   RoomName # from socket which can from ffmpeg
        participants
                        list   RoomName  # List or search for active rooms by name
                        get    --room|-r RoomName  --identity|-i ParticipantID  # Fetch metadata of a room participant
                        remove   # Remove a participant from a room
                        forward  # Forward a participant to a different room
                        move     # Move a participant to a different room
                        update   # Change the metadata and permissions for a room participant
        mute-track
        update-subscriptions
        send-data



lk egress
            start --type room-composite     <path/to/request.json> # Start room composite (recording of room UI)
            start --type track-composite    <path/to/request.json> # Start track composite (audio + video)
            start --type track              <path/to/request.json> # Start track egress (single audio or video track)


lk token create --join --room test-room --identity test-user --open meet
```
