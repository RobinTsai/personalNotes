# RTP 包

包格式由 Header 和 Payload 组成，Header 最少 12 Byte，最多 72 Byte；Payload 根据

```
Header 格式:

0               1               2               3               4
+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+
|V=2|P|X|  CC   |M|     PT      |   sequence number (16 bit)    |
+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+
| timestamp    32 bits                                          |
+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+
| synchronization source (SSRC) identifier   32 bits            |
+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+
| contributing source (CSRC) identifiers   N*32 bits            |
| ....                                                          |
+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+-+

V: Version, 占用 2 bit, RFC3550 规定设置为 2
P: Padding, 占用 1 bit, 1 表示报文末端有填充字节
X: Extension, 占用 1 bit, 1 表示固定包头后必须有扩展
CC: CSRC Count, 4 bit, 标识固定包头后 CSRC 数量，最大是 15
M: Marker, 1 bit, 由配置文件定义具体事件
PT: Payload Type, 7 bit, 标识负载类型, 编码算法、媒体类型、时钟频率和承载通道等
sequence number: 16 bit, 序列号，初始值随即随后按 1 递增, 用于重构数据并记录丢包量
timestamp: 32 bit, 初始值为随机数，每个采样递增 1; 如 8KHz 20ms 一个包时, 相邻 RTP 包递增 160
Synchronization Source, 32 bit, 标识 RTP 流的起源, 不同通道的不一样, 相同源的一直相同
CSRC list：贡献源列表, 标识 RTP 数据来源, 由混频器将所有贡献源的 SSRC 标识符放入此表中，数量 N 由 CC 决定, 最多 15 个
```

## RTP 和 PCM

RTP 在传输 PCMA、PCMU 编码的媒体的时候，会将 320 字节的 PCM 数据编码成 160 字节的 alaw 或 ulaw，然后加上 RTP 头 12Byte，所以每个包 172Byte。

要将 RTP 解析成 PCM，需要先拿到 RTP.payload，然后将 160 字节的 alaw 或 ulaw 解码成 320 字节的 PCM 数据。

// DecodeUlaw decodes u-law PCM data to 16bit LPCM
func DecodeUlaw(pcm []byte) []byte {
	lpcm := make([]byte, len(pcm)*2)
	for i, j := 0, 0; i < len(pcm); i, j = i+1, j+2 {
		frame := ulaw2lpcm[pcm[i]]
		lpcm[j] = byte(frame)
		lpcm[j+1] = byte(frame >> 8)
	}
	return lpcm
}

// DecodeAlaw decodes A-law PCM data to 16bit LPCM
func DecodeAlaw(pcm []byte) []byte {
	lpcm := make([]byte, len(pcm)*2)
	for i, j := 0, 0; i < len(pcm); i, j = i+1, j+2 {
		frame := alaw2lpcm[pcm[i]]
		lpcm[j] = byte(frame)
		lpcm[j+1] = byte(frame >> 8)
	}
	return lpcm
}

// 解码 RTP PCMA 负载为 PCM 数据
func decodePCMA(pcmaPayload []byte) []byte {
	pcmData := make([]byte, len(pcmaPayload)*2) // PCM数据是PCMA数据的两倍大小

	for i := 0; i < len(pcmaPayload); i++ {
		// PCMA解码算法，将8位PCMA样本转换为16位PCM样本
		sample := int(pcmaPayload[i]) << 8
		pcmData[i*2] = byte(sample & 0xFF)
		pcmData[i*2+1] = byte((sample >> 8) & 0xFF)
	}

	return pcmData
}

// 解码 RTP PCMU 负载为 PCM 数据
func decodePCMU(pcmuPayload []byte) []byte {
	pcmData := make([]byte, len(pcmuPayload)*2) // PCM数据是PCMU数据的两倍大小

	for i := 0; i < len(pcmuPayload); i++ {
		// PCMU解码算法，将8位PCMU样本转换为16位PCM样本
		sample := int(pcmuPayload[i]^0x80) << 8
		pcmData[i*2] = byte(sample & 0xFF)
		pcmData[i*2+1] = byte((sample >> 8) & 0xFF)
	}

	return pcmData
}
